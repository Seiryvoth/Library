\(\dfrac{a^m}{a^n}=a^{m-n}\) \(\log (a\times b)=\log a+\log b\)
\(a^m\times a^n=a^{m+n}\) \(\log\dfrac{a}{b}=\log a-\log b\)
\((a\times b)^n=a^nb^n\) \(\log a^n=n\log a\)
\((a^m)^n=a^{m\times n}\) \(\log_b a=\log_b x\times \log_x a\)
\(a^{\frac{1}{n}}=\sqrt[n]{a}\) \(\log_b a=\dfrac{\log_x a}{\log_x b}\)
\(\left(\dfrac{a}{b}\right)^n=\dfrac{a^n}{b^n},(b\neq 0)\) \(\log_a 1=0\)
\(a^0=1\) \( \log_a a=1\)
\(a^{-n}=\dfrac{1}{a^n},(a\neq 0)\) \(\log_e a=\ln a\)
\(a^{\frac{m}{n}}=\sqrt[n]{a^m}\) \(\ln e=1\)

\[\tan^2\alpha=\frac{1-\cos2\alpha}{1+\cos2\alpha} \]
\[\sin^3\alpha=\frac{1}{4}(3\sin\alpha-\sin3\alpha)\]
\[\cos^3\alpha=\frac{1}{4}(3\cos\alpha+\cos3\alpha)\]
\[\sin(\alpha\pm\beta)=\sin\alpha\cos\beta\pm\cos\alpha\sin\beta\]
\[\cos(\alpha\pm\beta)=\cos\alpha\cos\beta\mp\sin\alpha\sin\beta\]
\[\tan(\alpha\pm\beta)=\frac{\tan\alpha\pm\tan\beta}{1\mp\tan\alpha\tan\beta}\]
\[\cot(\alpha\pm\beta)=\frac{\cot\alpha\cot\beta\mp1}{\cot\beta\pm\cot\alpha}\]
\[\cos\alpha+\cos\beta=2\cos\frac{\alpha+\beta}{2}\cos\frac{\alpha-\beta}{2}\]
\[\cos\alpha-\cos\beta=-2\sin\frac{\alpha+\beta}{2}\sin\frac{\alpha-\beta}{2}\]
\[\sin\alpha+\sin\beta=2\sin\frac{\alpha+\beta}{2}\cos\frac{\alpha-\beta}{2}\]
