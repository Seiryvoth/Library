\[\sin\alpha-\sin\beta=2\sin\frac{\alpha-\beta}{2}\cos\frac{\alpha+\beta}{2}\]

\[\tan\alpha+\tan\beta=\frac{\sin(\alpha+\beta)}{\cos\alpha\cos\beta}\]
3 / 10
\[\tan\alpha-\tan\beta=\frac{\sin(\alpha-\beta)}{\cos\alpha\cos\beta}\]

\[\cot\alpha+\cot\beta=\frac{\sin(\alpha-\beta)}{\sin\alpha\sin\beta}\]
5 / 10
